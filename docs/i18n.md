# Internationalisation (i18n) — Reference

## Configuration

Declarer les langues dans `module.json` :
```json
"languages": ["fr", "en"]
```

La plateforme injecte `window.PLATFORM_LANG` et affiche un selecteur de langue dans le header.

## Fichier de traductions (translations.js)

```javascript
var TRANSLATIONS = {
    fr: {
        'titre': 'Analyseur de suggestions',
        'bouton.lancer': 'Lancer l\'analyse',
        'status.en_cours': 'Analyse en cours... {current}/{total}',
        'score.fort': 'Fort',
        'score.moyen': 'Moyen',
        'score.faible': 'Faible'
    },
    en: {
        'titre': 'Suggestion Analyzer',
        'bouton.lancer': 'Start analysis',
        'status.en_cours': 'Analyzing... {current}/{total}',
        'score.fort': 'Strong',
        'score.moyen': 'Medium',
        'score.faible': 'Weak'
    }
};
```

## Fonction de traduction (app.js)

```javascript
var langueActuelle = window.PLATFORM_LANG
    || new URLSearchParams(window.location.search).get('lg')
    || localStorage.getItem('langue')
    || 'fr';

function t(cle, params) {
    if (typeof TRANSLATIONS === 'undefined') return cle;
    var dict = TRANSLATIONS[langueActuelle] || TRANSLATIONS['fr'];
    var texte = (dict && dict[cle]) || (TRANSLATIONS['fr'] && TRANSLATIONS['fr'][cle]) || cle;
    if (params) {
        Object.keys(params).forEach(function(k) {
            texte = texte.replace(new RegExp('\\{' + k + '\\}', 'g'), params[k]);
        });
    }
    return texte;
}

// Traduction automatique des elements avec data-i18n
function traduirePage() {
    document.querySelectorAll('[data-i18n]').forEach(function(el) {
        el.innerHTML = t(el.getAttribute('data-i18n'));
    });
    document.querySelectorAll('[data-i18n-placeholder]').forEach(function(el) {
        el.placeholder = t(el.getAttribute('data-i18n-placeholder'));
    });
}
```

## HTML avec attributs i18n

```html
<h5 data-i18n="titre">Analyseur de suggestions</h5>
<button class="btn btn-primary" data-i18n="bouton.lancer">Lancer l'analyse</button>
<input type="text" data-i18n-placeholder="placeholder.requete" placeholder="Entrez votre requete">
```
